long long breweryAssign(int N, int M, int C, long long* brew, int* cursed) {
    long long low = 0, high = 0;

        long long base = brew[i];

            } else {
                segSum += brew[i];
                curseCnt = nextCurse;
                if (curseCnt > C) segSum *= 2;

int main() {
    ios::sync_with_stdio(false);

    cin >> M >> C;

    vector<long long> brew(N);
