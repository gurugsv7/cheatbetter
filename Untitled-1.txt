<template>
  <div id="app">
    <p><strong>Add a new contact:</strong></p>
    <div>
      <label>First Name: <input type="text" class="userFirstname" v-model="firstname" /></label> <br />
      <label>Last Name: <input type="text" class="userLastname" v-model="lastname" /></label> <br />
      <label>Phone: <input type="number" class="userPhone" v-model="phone" /></label> <br />
      <button class="submitButton" @click="addContact">Add New Contact</button>
    </div>
    <div>
      <table id="phoneBookItems" class="informationTable">
        <tr v-for="item in sortedItems" :key="item.lastname + item.firstname">
          <td>{{ item.firstname }}</td>
          <td>{{ item.lastname }}</td>
          <td>{{ item.phone }}</td>
        </tr>
      </table>
    </div>
  </div>
</template>

<script>
export default {
  name: "App",
  data() {
    return {
      firstname: "Coder",
      lastname: "Byte",
      phone: "8885559999",
      items: []
    };
  },
  computed: {
    sortedItems() {
      return this.items.slice().sort(function(a, b) {
        return a.lastname.localeCompare(b.lastname);
      });
    }
  },
  methods: {
    addContact() {
      this.items.push({
        firstname: this.firstname,
        lastname: this.lastname,
        phone: this.phone
      });
    }
  }
};
</script>
